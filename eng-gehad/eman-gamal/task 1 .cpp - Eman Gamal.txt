#include <iostream>
using namespace std;

struct Node {
    int data;
    Node* next;
};

void printList(Node* head)
{
    for (Node* temp = head; temp != NULL; temp = temp->next)
    {
        cout << temp->data << " ";
    }
}

int main()
{
    Node* head = new Node{6,NULL};
    head->next = new Node{10,NULL};
    head->next->next = new Node{8,NULL};
    head->next->next->next = new Node{15,NULL};

    printList(head);

    return 0;
}
