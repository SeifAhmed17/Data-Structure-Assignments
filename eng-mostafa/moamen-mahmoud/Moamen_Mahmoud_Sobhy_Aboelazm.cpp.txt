void Task5() { // Reverse number using stack
    int number = 123456 ; // cin >> number;
    Stack<int> stack;
    while (number) {
        stack.push(number % 10);
        number /= 10;
    }

    int k = 1;
    while (!stack.is_empty()) {
        number += stack.peek()*k;
        k *= 10;
        stack.pop();
    }
    cout << number << endl;
}

void Task6() { // Valid Parentheses
    string word = "{[()]}";  cin >> word;
    Stack<char> stack;

    for (char c : word) {
        if (c == '{' || c == '(' || c == '[') stack.push(c);
        else {
            char top = stack.peek();

            if ( (c == '}' && top == '{') || (c == ']' && top == '[') || (c == ')' && top == '(') ) stack.pop();
            else return void (cout << "The Word is Invalid!");
        }
    }
    cout << "The Word is Valid!" << endl;
}

void Task8() { // Remove Duplicates
    int n = 7; // cin >> n;
    Stack<int> st;
    for (int i = 1; i <= n; i++) {
        int x ; cin >> x;
        if (i == 1) st.push(x);
        if (i > 1 && st.peek() != x) st.push(x);
    }
    Stack<int> tmp;
    while (!st.is_empty()) {
        tmp.push(st.peek());
        st.pop();
    }
    while (!tmp.is_empty()) {
        cout << tmp.peek() << " ";
        tmp.pop();
    }
}

void Task10() { // Check if palindrome
    Stack<int> st1,st2 , tmp;
    tmp.push(1);
    tmp.push(2);
    tmp.push(3);
    tmp.push(2);
    tmp.push(1);

    while (!tmp.is_empty()) {
        st2.push(tmp.peek());
        st1.push(tmp.peek());
        tmp.pop();
    }

    st2.reverse();

    while (!st2.is_empty()) {
        if ( st2.peek() != st1.peek()) return void (cout << "No Palindrome");
        st1.pop();
        st2.pop();
    }

    cout << "Palindrome";
}
